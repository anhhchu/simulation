def compute_avg_daily_cost(Q, T, d, s, N, qi):
    """
    Compute the average daily cost of a water station based on the given input parameters.

    Args:
        Q (float): Delivery quantity in gallons.
        T (int): Time between deliveries in days.
        d (float): Cost of delivery in dollars.
        s (float): Storage cost per gallon per day in dollars.
        N (int): Total simulation length in days.
        qi (func): func to calculate the distribution of demand


    Returns:                    
        float: average daily cost 
    """

    I = 0 # current inventory
    C = 0 # total running cost
    cycles = N // T # number of delivery cycles if deliver every T days
    
    shortage_days = 0

    for cycle in range(cycles):
        # add delivery gallons to inventory
        I += Q 
        # add cost of deliver to total Cost
        C += d 

        # each day between inventory cycle
        for day in range(T): 
            # demand for the day
            ri = qi()
            # substract demand from inventory
            if I >= ri: 
                I -= ri
            else: 
                I = 0
                shortage_days += 1
            # add storage cost to total cost
            C += I * s 

    # remaining days not accounted for in the cycles
    for day in range(N%T): 
        ri = qi()
        # substract demand from inventory
        if I >= ri: 
            I -= ri
        else: 
            I = 0
            shortage_days += 1
        # add storage cost to total cost
        C += I * s 

    return C/N, shortage_days

# Represent the Cumulative hist with linear splines
# find m and b for the piecewise linear interpolations q = mx + b
m = np.zeros((demand.size - 1))
b = np.zeros((demand.size - 1))

for i in np.arange(demand.size - 1):
    m[i] = (q[i+1] - q[i]) / (demand[i+1] - demand[i])
    b[i] = q[i] - m[i] * demand[i]

# Inverse function x = (q - b)/m or x = q/m - b/m
minv = 1.0/m 
binv = -b/m

# Compute inverses of linear splines to compute daily demand as a function of random number between [0,1]
import bisect
def qi_hist():
    """
    Return the demand based on historical data using inverse linear splines function
    """
    p = np.random.uniform()
    interval = bisect.bisect_right(q, p) 
    random_demand = minv[interval - 1]*p + binv[interval - 1]
    return random_demand
